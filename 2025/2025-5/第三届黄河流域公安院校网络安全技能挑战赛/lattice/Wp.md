def gen(q, n, N, sigma):
    t = np.random.randint(0, high=q // 2, size=n)  # 随机私钥部分
    s = np.concatenate([np.ones(1, dtype=np.int32), t])  # 完整私钥 s = [1, t1, t2, ..., tn]
    
    A = np.random.randint(0, high=q // 2, size=(N, n))  # 随机矩阵 A
    e = np.round(np.random.randn(N) * sigma**2).astype(np.int32) % q  # 高斯噪声 e
    
    b = ((np.dot(A, t) + e).reshape(-1, 1)) % q  # 计算 b = A·t + e (mod q)
    P = np.hstack([b, -A])  # 公钥 P = [b | -A]
    
    return P, s

def enc(P, M, q):
    N = P.shape[0]
    n = len(M)
    r = np.random.randint(0, 2, (n, N))  # 随机比特矩阵 r ∈ {0,1}^(n×N)
    
    Z = np.zeros((n, P.shape[1]), dtype=np.int32)
    Z[:, 0] = 1  # Z = [1, 0, 0, ..., 0]（用于嵌入消息）
    
    C = np.zeros((n, P.shape[1]), dtype=np.int32)
    for i in range(n):
        C[i] = (np.dot(P.T, r[i]) + (np.floor(q / 2) * Z[i] * M[i])) % q
    return C

# 求flag

$$
\sin{(iv)} + leak\times \cos{(iv)} = 0
$$

等价于
$$
\tan{(iv)} = -leak
$$

$$
\therefore iv = \arctan{(-leak)} + 2k\pi
$$

很经典的板子题，造格
$$
\begin{pmatrix}
iv & -1 & k
\end{pmatrix}
\begin{pmatrix}
1 & 0 & 1\\
0 & 2^t & \arctan{-leak}\\
0 & 0 & 2\pi
\end{pmatrix}
=
\begin{pmatrix}
iv & -2^k & 0
\end{pmatrix}
$$
